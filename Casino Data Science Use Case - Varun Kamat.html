def exploratory_data_analysis(df):
    """
    Performs complex and intuitive exploratory data analysis (EDA) on the provided DataFrame. 
    
    Visualizations:
    1. Customer Lifetime Value by Acquisition Source
       Assumes 'sum_ngr' is the net gain/loss for the casino from the player. It plots the total net gain/loss by acquisition source.
    
    2. Average Deposit and Bonus Cost Ratio by Country
       Helps understand the effectiveness of bonuses contributing to deposits in different countries by plotting the deposit to bonus cost ratio.
    
    3. Monthly Betting Activity Trend
       Shows the trend of betting activity over months which can inform about peak times or seasonal trends in betting activity.
    
    4. Customer Retention Over Time
       Investigates how long players stay active after their first deposit, which can inform strategies for improving player retention.
    
    5. Distribution of Sum of Bonuses
       A histogram of 'sum_bonus_cost' values with a log scale, including mean and median lines to show the central tendency.
    
    6. Heatmap of Feature Correlations
       Displays a heatmap of correlations between numerical features to identify relationships.
    
    7. Unique Value Counts for Categorical Variables
       Prints a table showing the number of unique values for each categorical variable, aiding in deciding the type of encoding to use.
    
    
    Parameters:
    - df: pd.DataFrame, the DataFrame on which EDA is to be performed.
    """
    
    # Visualization 1: Customer Lifetime Value by Acquisition Source
    # We assume 'sum_ngr' is the net gain/loss for the casino from the player.
    df['customer_lifetime_value'] = df.groupby('customer_acquisition_source')['sum_ngr'].transform('sum')
    plt.figure(figsize=(10, 6))
    sns.barplot(data=df.drop_duplicates('customer_acquisition_source'), x='customer_acquisition_source', y='customer_lifetime_value')
    plt.title('Customer Lifetime Value by Acquisition Source')
    plt.xticks(rotation=45)
    plt.xlabel('Acquisition Source')
    plt.ylabel('Total Net Gain/Loss')
    plt.show()
    
    # Visualization 2: Average Deposit and Bonus Cost Ratio by Country
    # This visualization helps understand if the bonuses are effectively contributing to deposits in different countries.
    df['avg_deposit'] = df.groupby('customer_country')['sum_deposit'].transform('mean')
    df['avg_bonus_cost'] = df.groupby('customer_country')['sum_bonus_cost'].transform('mean')
    df['deposit_bonus_ratio'] = df['avg_deposit'] / df['avg_bonus_cost']
    country_data = df.drop_duplicates('customer_country')[['customer_country', 'deposit_bonus_ratio']].dropna()
    plt.figure(figsize=(10, 6))
    sns.barplot(data=country_data, x='customer_country', y='deposit_bonus_ratio', order=country_data.sort_values('deposit_bonus_ratio', ascending=False)['customer_country'])
    plt.title('Average Deposit and Bonus Cost Ratio by Country')
    plt.xticks(rotation=90)
    plt.xlabel('Country')
    plt.ylabel('Deposit/Bonus Cost Ratio')
    plt.show()
    
    # Visualization 3: Monthly Betting Activity Trend
    df['transaction_month'] = pd.to_datetime(df['transaction_date']).dt.month
    monthly_activity = df.groupby('transaction_month')['sum_bets'].sum().reset_index()
    
    plt.figure(figsize=(12, 6))
    sns.lineplot(data=monthly_activity, x='transaction_month', y='sum_bets', marker='o')
    plt.title('Trend of Betting Activity Over Months')
    plt.xlabel('Month')
    plt.ylabel('Total Bets')
    plt.xticks(ticks=np.arange(1, 13), labels=[f'Month {i}' for i in range(1, 13)])  # Assuming 1-12 are months of the year
    plt.grid(True)
    plt.show()
    
    # Visualization 4: Customer Retention Over Time
    # Investigating how long players stay active after their first deposit can inform retention strategies.
    df['days_since_first_deposit'] = (pd.to_datetime(df['transaction_date']) - pd.to_datetime(df['customer_first_deposit_date'])).dt.days
    retention_over_time = df.groupby('days_since_first_deposit').size()
    plt.figure(figsize=(10, 6))
    sns.lineplot(data=retention_over_time)
    plt.title('Customer Retention Over Time')
    plt.xlabel('Days Since First Deposit')
    plt.ylabel('Number of Active Players')
    plt.show()
    
    # Visualization 5: Histogram of sum_bonus_cost values distribution trend
    # Histogram of sum_bonus_cost values
    plt.figure(figsize=(14, 6))
    plt.subplot(1, 2, 1)
    sns.histplot(df['sum_bonus_cost'], bins=40, log_scale=False, color='blue', alpha=0.7)
    mean_value = df['sum_bonus_cost'].mean()
    median_value = df['sum_bonus_cost'].median()
    plt.axvline(x=mean_value, color='red', linestyle='--', label=f'Mean = {mean_value:.2f}')
    plt.axvline(x=median_value, color='green', linestyle='--', label=f'Median = {median_value:.2f}')
    plt.title('Histogram of Sum of Bonuses')
    plt.xlabel('Sum of Bonuses')
    plt.ylabel('Frequency')
    plt.legend()
    plt.show()
    
    # Heatmap of correlations
    numeric_df = df.select_dtypes(include=[np.number])
    plt.figure(figsize=(10, 8))
    sns.heatmap(numeric_df.corr(), annot=True, fmt='.2f', cmap='coolwarm')
    plt.title('Heatmap of Feature Correlations')
    plt.show()
    
    # View distributions of sum_bets and sum_wins.
    distribution_metrics = df[['sum_bets', 'sum_wins']].describe()
    print(tabulate(distribution_metrics, headers='keys', tablefmt='pretty', floatfmt=".2f"))
    
    # Printing the unique value count for categorical variables to decide on the type of encoding
    # Filter for categorical variables
    categorical_vars = df.select_dtypes(include=['object', 'category']).columns
    # Calculate the number of unique values for each categorical variable
    unique_counts = {var: df[var].nunique() for var in categorical_vars}
    # Convert to a format suitable for tabulate
    table = [[var, unique_counts[var]] for var in unique_counts]
    headers = ['Categorical Variable', 'Number of Unique Values']
    # Print the table using tabulate
    print(tabulate(table, headers=headers, tablefmt='grid'))
    
    # Return the DataFrame with added analysis columns
    return df

3. Data Preprocessing for Classification

The preprocess_data_for_classification function meticulously prepares a DataFrame for binary classification tasks. It ensures numerical features are normalized and a continuous variable is transformed into a binary target, which is crucial for machine learning models to accurately interpret the data.

Function Overview

  • Input: Receives a DataFrame df with the dataset to be preprocessed.
  • Output: Yields a tuple (X, y) where X is the processed feature matrix, and y is the binary target variable.

Preprocessing Steps

1. Target Variable Creation
  • Description: Converts the sum_bonus_cost variable from continuous to binary (high=1, low=0).
  • Importance: Facilitates the use of binary classification algorithms by simplifying the outcome into two classes.
2. Feature Engineering

Enhances the dataset with new features that could provide deeper insights into customer behavior:

  • bet_win_ratio: Created by dividing the total bets by total wins, adding a small epsilon to avoid division by zero. It captures a player's risk-reward profile.
  • days_since_first_deposit: Measures the number of days since a customer's first deposit, indicating engagement duration.
  • days_for_first_deposit: Calculates the time taken for a customer to make their first deposit after account creation, potentially reflecting initial engagement.
  • days_since_account_creation: The time elapsed since the account was created, which could correlate with customer loyalty.
  • transaction_month: Extracts the month from the transaction date, useful for identifying seasonal patterns in betting activity.

Each new feature is designed to reveal different aspects of player behavior that could influence their response to bonuses. By including these engineered features, the model can better classify which players might increase their betting activity as a result of receiving a bonus.

3. Redundant Feature Removal
  • Description: Drops features that are redundant or have been used to engineer more meaningful attributes.
  • Importance: Streamlines the feature set to reduce overfitting and improve model interpretability.
4. Categorical Variable Encoding
  • Description: Applies label encoding to transform categorical variables into a numeric format suitable for modeling.
  • Importance: Ensures all model inputs are quantitative, which is necessary for most machine learning algorithms.
5. Feature and Target Selection
  • Description: Delineates the features and the target variable, preparing them for the machine learning pipeline.
  • Importance: Clearly separates the predictors from the outcome, a prerequisite for supervised learning models.
6. Feature Normalization
  • Description: Normalizes features using StandardScaler to have a mean of zero and a variance of one.
  • Importance: Provides a level playing field for all features, crucial for models that are sensitive to the scale of inputs.

Conclusion

This preprocessing pipeline, with its targeted feature creation and careful data preparation, equips the classification model to discern between players who are likely to use bonuses to place more bets and those who are not. The features engineered provide a comprehensive view of player behavior, which is instrumental in predicting their response to promotional strategies, thereby enabling more effective and profitable bonus allocations.

def preprocess_data_for_classification(df):
    """
    Preprocesses the provided DataFrame for a classification task including normalization.
    
    Parameters:
    - df: pd.DataFrame, the DataFrame to be preprocessed.
    
    Returns:
    - X: pd.DataFrame, the processed feature matrix.
    - y: pd.Series, the binary target variable.
    """
    # Creation of the Target Variable
    # Convert 'sum_bonus_cost' into binary target variable (high=1, low=0)
    df['bonus_class'] = (df['sum_bonus_cost'] > df['sum_bonus_cost'].min()).astype(int)
    
    # Feature Engineering
    # 1. Created a 'deposit_bonus_ratio' in the exploratory phase
    #df['deposit_bonus_ratio'] = df['avg_deposit'] / df['avg_bonus_cost']
    
    # 2. Create a new feature 'bet_win_ratio'
    epsilon = 1e-8  # A small constant to avoid division by zero
    df['bet_win_ratio'] = df['sum_bets'] / (df['sum_wins'] + epsilon)
    # 3. Create a new feature days_since_first_deposit
    df['days_since_first_deposit'] = (pd.to_datetime(df['transaction_date']) - pd.to_datetime(df['customer_first_deposit_date'])).dt.days
    # 4. Create a new feature days_for_first_deposit
    df['days_for_first_deposit'] = (pd.to_datetime(df['customer_first_deposit_date']) - pd.to_datetime(df['customer_account_created_date'])).dt.days
    # 5. Create a new feature days_since_account_creation
    df['days_since_account_creation'] = (pd.to_datetime(df['transaction_date'])-pd.to_datetime(df['customer_account_created_date'])).dt.days
    # 6. Create a new feature transaction_month to isolate the month from the trasaction_date
    df['transaction_month'] = pd.to_datetime(df['transaction_date']).dt.month
    
    # 3. Drop redundant features as they were used to create new more meaningful features
    columns_to_drop = ['sum_bonus_cost','sum_bets','sum_wins','customer_id','transaction_date','customer_first_deposit_date','customer_account_created_date', 'customer_id', 'avg_deposit', 'avg_bonus_cost']
    df = df.drop(columns_to_drop, axis=1)
    
    # 4. Identify categorical variables and encode them
    categorical_cols = df.select_dtypes(include=['object', 'category']).columns
    # Apply label encoding to categorical variables
    label_encoder = LabelEncoder()
    for col in categorical_cols:
        df[col] = label_encoder.fit_transform(df[col]).astype(float)
    
    # remove infinities due to 29 infinities in deposit_bonus_ratio
    df = df.replace([np.inf, -np.inf], np.nan).dropna(axis=0, how='any')
    
    # 5. Select features and target
    features = df[['sum_deposit', 'bet_win_ratio', 'sum_ngr','days_since_first_deposit', 'days_for_first_deposit','days_since_account_creation', 'deposit_bonus_ratio', 'transaction_month']]
    target = df['bonus_class']
    
    # 6. Normalize features
    scaler = StandardScaler()
    features_normalized = scaler.fit_transform(features)
    
    return features, target

The modeling_and_evaluation function is designed to train multiple classification models on a given dataset, evaluate their performance through the AUC-ROC metric, visualize their AUC-ROC curves, and compile the predictions into a DataFrame.

Function Overview

  • Input: The function accepts training and testing sets (X_train, X_test, y_train, y_test) as inputs.
  • Output: Returns a DataFrame df_predictions containing the predicted classes for each model.

Process Description

1. Model Initialization
  • Models Included: The function initializes and trains several classification models, including Random Forest, Gradient Boosting, and Support Vector Machine (SVM) with probability estimates enabled for AUC calculation.
  • SVM Specifics: The SVM model is specifically noted for having probability estimates enabled, a requirement for calculating the AUC-ROC curve.
2. Model Training and Prediction
  • Training: Each model is trained on the X_train and y_train datasets.
  • Prediction: The function then predicts class probabilities (y_pred_proba) and classes (y_pred_class) on the X_test set.
3. Performance Evaluation and Visualization
  • AUC-ROC Metric: The Area Under the Receiver Operating Characteristic (AUC-ROC) curve is calculated for each model, serving as a performance metric.
  • ROC Curve Visualization: A plot of the ROC curve for each model is generated, comparing their performance visually. This includes a diagonal line representing the performance of a random classifier for reference.
4. Compiling Predictions
  • DataFrame of Predictions: Predicted classes from each model are added to df_predictions, creating a consolidated view of all model predictions.

Significance of the AUC-ROC Curve

  • Performance Indicator: The AUC-ROC curve is a crucial metric for evaluating classification model performance, especially in binary classification tasks. It illustrates a model's ability to distinguish between the classes across different thresholds.
  • Comparison of Models: By visualizing the ROC curves of different models together, we can compare their performance in terms of sensitivity (True Positive Rate) and specificity (False Positive Rate) at various threshold settings.
  • Optimal Model Selection: The model with the highest AUC value can be considered as having the best performance among the evaluated models, guiding the selection of the most appropriate model for the task.

Conclusion

The modeling_and_evaluation function streamlines the process of training, evaluating, and comparing multiple classification models on a given dataset. By focusing on the AUC-ROC curve as a comprehensive performance metric, it aids in identifying the most effective model, ultimately contributing to more informed decision-making in predictive modeling tasks.

def modeling_and_evaluation(X_train, X_test, y_train, y_test):
    models = {
        'Random Forest': RandomForestClassifier(n_estimators=100, random_state=42),
        'Gradient Boosting': GradientBoostingClassifier(n_estimators=100, random_state=42),
        'Support Vector Machine': SVC(probability=True, random_state=42),
        'LightGBM': LGBMClassifier(is_unbalance=True, class_weight='balanced', random_state=42)
    }
    
    df_predictions = pd.DataFrame()
    confusion_matrices = {}

    # Initialize plot for ROC curve
    plt.figure(figsize=(10, 8))
    plt.plot([0, 1], [0, 1], 'k--', label='Random chance')

    for name, model in models.items():
        model.fit(X_train, y_train)
        
        if hasattr(model, "predict_proba"):
            y_pred_proba = model.predict_proba(X_test)[:, 1]
        else:  # Use decision function if predict_proba is not available
            y_pred_proba = model.decision_function(X_test)
            y_pred_proba = (y_pred_proba - y_pred_proba.min()) / (y_pred_proba.max() - y_pred_proba.min())
        
        y_pred_class = model.predict(X_test)
        df_predictions[f'{name}_class'] = y_pred_class
        
        auc_score = roc_auc_score(y_test, y_pred_proba)
        fpr, tpr, _ = roc_curve(y_test, y_pred_proba)
        plt.plot(fpr, tpr, label=f'{name} (AUC = {auc_score:.2f})')

        # Calculate and store confusion matrix
        cm = confusion_matrix(y_test, y_pred_class)
        confusion_matrices[name] = cm

    # Finalize ROC curve plot
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.title('ROC Curves for All Models')
    plt.legend(loc='lower right')
    plt.show()

    # Plot feature importance for LightGBM in a separate figure
    if 'LightGBM' in models:
        plt.figure(figsize=(10, 6))
        model = models['LightGBM']
        feature_importance = model.feature_importances_
        sorted_idx = np.argsort(feature_importance)[-6:]
        plt.barh(range(len(sorted_idx)), feature_importance[sorted_idx], align='center')
        plt.yticks(range(len(sorted_idx)), [X_train.columns[i] for i in sorted_idx])
        plt.title('Feature Importance')
        plt.xlabel('Feature Importance Score')
        plt.ylabel('Features')
        plt.show()

    # Print confusion matrices in a prettier table
    print("\nConfusion Matrix Results:")
    for model_name, cm in confusion_matrices.items():
        cm_df = pd.DataFrame(cm, index=['Actual Negative', 'Actual Positive'], columns=['Predicted Negative', 'Predicted Positive'])
        print(f"\n{model_name} Confusion Matrix:")
        print(tabulate(cm_df, headers='keys', tablefmt='pretty'))
    
    return df_predictions

def main(file_path):

app = Flask(__name__)

# Define the prediction endpoint. The route() decorator tells Flask that this function should be triggered for the URL path '/predict' with the HTTP method POST.
@app.route('/predict', methods=['POST'])
def predict():
    # Parse the JSON payload from the incoming POST request.
    # The force=True argument ensures that only JSON is expected.
    data = request.get_json(force=True)
    
    # Check if the incoming JSON data is a list (multiple records) or a dictionary (single record).
    # If it's a single record, it's wrapped in a list to create a uniform DataFrame structure.
    # This is important because the preprocessing function expects a DataFrame input.
    df = pd.DataFrame(data if isinstance(data, list) else [data])
    
    # Preprocess the data using the same function used during model training.
    # This ensures that the new data is transformed in the same way as the training data.
    # The preprocess_data_for_classification function handles both training (with target) and prediction (without target) cases, hence the slicing to ignore the target in the output.
    X, _ = preprocess_data_for_classification(df)
